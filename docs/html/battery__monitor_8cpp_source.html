123BatteryStatus BatteryMonitor::readBatteryStatus()
124{
125 BatteryStatus status;
126
127 if (!deviceFound_)
128 {
129 throw std::runtime_error("Battery monitor not initialized");
130 }
131
132 try
133 {
134 // Read voltage (register 0x08)
135 uint16_t rawVoltage = readRegister(0x08, 2);
136 status.voltage = convertVoltage(rawVoltage);
137
138 // Read current (register 0x10)
139 int16_t rawCurrent = static_cast<int16_t>(readRegister(0x10, 2));
140 status.current = convertCurrent(rawCurrent);
141
142 // Read average current (register 0x0A)
143 int16_t rawAvgCurrent = static_cast<int16_t>(readRegister(0x0A, 2));
144 status.averageCurrent = convertCurrent(rawAvgCurrent);
145
146 // Read remaining capacity (register 0x04)
147 uint16_t rawCapacity = readRegister(0x04, 2);
148 status.remainingCapacity = static_cast<float>(rawCapacity * 2); // mAh
149
150 // Read State of Charge (register 0x02)
151 status.stateOfCharge = static_cast<int>(readRegister(0x02, 1));
152
153 // Read State of Health (register 0x2E)
154 status.stateOfHealth = static_cast<int>(readRegister(0x2E, 1));
155
156 // Read temperatures
157 uint16_t rawCellTemp = readRegister(0x0C, 1);
158 status.cellTemperature = convertTemperature(rawCellTemp);
159
160 uint16_t rawBoardTemp = readRegister(0x2A, 1);
161 status.boardTemperature = convertTemperature(rawBoardTemp);
162
163 // Read cycle count (register 0x2C)
164 status.cycleCount = static_cast<int>(readRegister(0x2C, 1));
165
166 // Read average time to empty (register 0x18)
167 status.averageTimeToEmpty = static_cast<int>(readRegister(0x18, 2));
168
169 // Perform safety checks
170 if (!isBatterySafe(status))
171 {
172 throw std::runtime_error("Battery safety limits exceeded");
173 }
174
175 return status;
176 }
177 catch (const std::exception &e)
178 {
179 throw std::runtime_error("Failed to read battery status: " + std::string(e.what()));
180 }
181}
