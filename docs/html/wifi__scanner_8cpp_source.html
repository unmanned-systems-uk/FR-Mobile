195void WiFiScanner::processProbeRequest(const uint8_t *packet, size_t length, int rssi)
196{
197 if (!scanning_.load())
198 {
199 LOG_DEBUG(COMPONENT_NAME, "Ignoring probe request - scanner not active");
200 return;
201 }
202
203 if (!isValidProbeRequest(packet, length))
204 {
205 LOGF_DEBUG(COMPONENT_NAME, "Invalid probe request packet (length: %zu, type: 0x%02x)",
206 length, packet ? packet[0] : 0);
207 return;
208 }
209
210 try
211 {
212 ProbeRequest request;
213 request.dataType = "Wi-Fi";
214 request.timestamp = timeManager_->getCurrentDateTime();
215 request.source = "wifi";
216 request.rssi = rssi;
217 request.packetLength = static_cast<int>(length);
218 request.macAddress = extractMAC(packet);
219 request.payload = packetToHexString(packet, length);
220
221 // Check if MAC should be ignored
222 if (Utils::isIgnoredMAC(request.macAddress))
223 {
224 LOGF_DEBUG(COMPONENT_NAME, "Ignoring probe request from filtered MAC: %s",
225 request.macAddress.c_str());
226 return;
227 }
228
229 // Save the probe request
230 saveProbeRequest(request);
231
232 // Call user callback if registered
233 if (probeCallback_)
234 {
235 try
236 {
237 probeCallback_(request);
238 }
239 catch (const std::exception &e)
240 {
241 LOG_ERROR(COMPONENT_NAME, "Error in probe callback: " + std::string(e.what()));
242 }
243 }
244
245#ifdef DEBUG_WIFI_SCAN
246 LOGF_DEBUG(COMPONENT_NAME, "Probe Request: MAC=%s, RSSI=%d, Length=%d, Time=%s",
247 request.macAddress.c_str(), request.rssi, request.packetLength,
248 request.timestamp.c_str());
249#endif
250 }
251 catch (const std::exception &e)
252 {
253 LOG_ERROR(COMPONENT_NAME, "Error processing probe request: " + std::string(e.what()));
254 }
255}

257bool WiFiScanner::isValidProbeRequest(const uint8_t *packet, size_t length)
258{
259 if (!packet)
260 {
261 LOG_DEBUG(COMPONENT_NAME, "Null packet received");
262 return false;
263 }
264
265 if (length < MIN_PACKET_SIZE)
266 {
267 LOGF_DEBUG(COMPONENT_NAME, "Packet too small: %zu bytes (minimum: %zu)",
268 length, MIN_PACKET_SIZE);
269 return false;
270 }
271
272 if (packet[0] != PROBE_REQUEST_TYPE)
273 {
274 LOGF_DEBUG(COMPONENT_NAME, "Wrong packet type: 0x%02x (expected: 0x%02x)",
275 packet[0], PROBE_REQUEST_TYPE);
276 return false;
277 }
278
279 return true;
280}

282std::string WiFiScanner::extractMAC(const uint8_t *packet)
283{
284 if (!packet)
285 {
286 LOG_ERROR(COMPONENT_NAME, "Cannot extract MAC from null packet");
287 return "";
288 }
289
290 try
291 {
292 char macStr[18];
293 std::snprintf(macStr, sizeof(macStr), "%02x:%02x:%02x:%02x:%02x:%02x",
294 packet[MAC_ADDRESS_OFFSET], packet[MAC_ADDRESS_OFFSET + 1],
295 packet[MAC_ADDRESS_OFFSET + 2], packet[MAC_ADDRESS_OFFSET + 3],
296 packet[MAC_ADDRESS_OFFSET + 4], packet[MAC_ADDRESS_OFFSET + 5]);
297 return std::string(macStr);
298 }
299 catch (const std::exception &e)

306std::string WiFiScanner::packetToHexString(const uint8_t *packet, size_t length)
307{
308 if (!packet || length == 0)
309 {
310 LOG_WARNING(COMPONENT_NAME, "Cannot convert null/empty packet to hex string");
311 return "";
312 }
313
314 try
315 {
316 std::string result;
317 result.reserve(length * 3);
318
319 for (size_t i = 0; i < length; ++i)
320 {
321 char hex[4];
322 std::snprintf(hex, sizeof(hex), "%02x ", packet[i]);
323 result += hex;
324 }
325
326 // Remove trailing space
327 if (!result.empty())
328 {
329 result.pop_back();
330 }
331
332 return result;
333 }
334 catch (const std::exception &e)
335 {
336 LOG_ERROR(COMPONENT_NAME, "Error converting packet to hex: " + std::string(e.what()));
337 return "";
338 }
