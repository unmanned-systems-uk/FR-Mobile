312std::string Utils::escapeJSONString(const std::string &input)
313{
314 std::string output;
315 output.reserve(input.length() * 2);
316
317 for (char c : input)
318 {
319 switch (c)
320 {
321 case '"':
322 output += "\\\"";
323 break;
324 case '\\':
325 output += "\\\\";
326 break;
327 case '\b':
328 output += "\\b";
329 break;
330 case '\f':
331 output += "\\f";
332 break;
333 case '\n':
334 output += "\\n";
335 break;
336 case '\r':
337 output += "\\r";
338 break;
339 case '\t':
340 output += "\\t";
341 break;
342 default:
343 if (c < 0x20 || c > 0x7E)
344 {
345 char buf[7];
346 std::snprintf(buf, sizeof(buf), "\\u%04x", static_cast<unsigned char>(c));
347 output += buf;
348 }
349 else
350 {
351 output += c;
352 }
353 break;
354 }
355 }
356
357 return output;
358}

360long Utils::convertToTotalMinutes(const std::string &dateTimeStr)
361{
362 try
363 {
364 int year = std::stoi(dateTimeStr.substr(0, 4));
365 int month = std::stoi(dateTimeStr.substr(5, 2));
366 int day = std::stoi(dateTimeStr.substr(8, 2));
367 int hour = std::stoi(dateTimeStr.substr(11, 2));
368 int minute = std::stoi(dateTimeStr.substr(14, 2));
369
370 // Simplified calculation (assuming 30 days per month, 365 days per year)
371 long totalMinutes = minute + (hour * 60) + (day * 24 * 60) +
372 (month * 30 * 24 * 60) + (year * 365 * 24 * 60);
373
374 return totalMinutes;
375 }
376 catch (const std::exception &e)
377 {
378 printf("Error converting time to minutes: %s\n", e.what());
379 return 0;
