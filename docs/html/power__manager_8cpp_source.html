590// Get power statistics
591PowerStats PowerManager::getPowerStats() const
592{
593 PowerStats stats = powerStats_;
594
595 // Calculate current session stats
596 auto now = std::chrono::steady_clock::now();
597 auto sessionDuration = std::chrono::duration_cast<std::chrono::microseconds>(
598 now - bootTime_)
599 .count();
600
601 // Update totals
602 stats.totalActiveTime = accumulatedActiveTime_;
603 stats.totalSleepTime = accumulatedSleepTime_;
604
605 // Calculate averages
606 if (stats.sleepCycles > 0)
607 {
608 stats.averageSleepDuration = static_cast<float>(stats.totalSleepTime) /
609 static_cast<float>(stats.sleepCycles) / 1000000.0f;
610 }
611
612 // Calculate duty cycle
613 uint64_t totalTime = stats.totalActiveTime + stats.totalSleepTime;
614 if (totalTime > 0)
615 {
616 stats.dutyCycle = (static_cast<float>(stats.totalActiveTime) /
617 static_cast<float>(totalTime)) *
618 100.0f;
619 }
620
621 return stats;
622}
